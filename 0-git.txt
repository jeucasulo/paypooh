Git global setup
git config --global user.name "Jeú Guimarães Conrado Junior"
git config --global user.email "jeucasulo@hotmail.com"

Create a new repository
git clone git@gitlab.com:freelasweb/paypooh.git
cd paypooh
touch README.md
git add README.md
git commit -m "add README"
git push -u origin master

Existing folder
cd existing_folder
git init
git remote add origin git@gitlab.com:freelasweb/paypooh.git
git add .
git commit -m "Initial commit"
git push -u origin master

Existing Git repository
cd existing_repo
git remote rename origin old-origin
git remote add origin git@gitlab.com:freelasweb/paypooh.git
git push -u origin --all
git push -u origin --tags

…or create a new repository on the command line
echo "# paypooh" >> README.md
git init
git add README.md
git commit -m "first commit"
git remote add origin https://github.com/jeucasulo/paypooh.git
git push -u origin master
…or push an existing repository from the command line
git remote add origin https://github.com/jeucasulo/paypooh.git
git push -u origin master
